模板

提炼出排序系列算法的通用模板，基本的比较、展示、获取都涵盖其中。真实情况下一般是对某些对象进行排序，所以我设计的排序的对象不再是常见的int形，而是实现Comparable[]的任何对象。

package uestc.zhangkx.sort;

import org.jetbrains.annotations.Contract;
import org.jetbrains.annotations.NotNull;
import org.springframework.stereotype.Component;


import java.util.Arrays;
import java.util.Calendar;
import java.util.Collection;

/**
 * 排序算法的模板
 *
 * @author zhangkx
 */
@Component
public class SortTemplate {

    public static Comparable[] al = {12, 4443,9999,23,2,0, 435, 23, 213, 54, 22, 76, 3, 0, 443, 25, 6, 23, 125, 678, 987, 55, 33, 42, 75, 23, 1, 9, 6654};

    public static Comparable[] aux = null;
    public int count = 0;

    public void sort(Comparable[] a) {

    }

    @Contract(pure = true)
    public static boolean less(@NotNull Comparable v, @NotNull Comparable w) {
        return v.compareTo(w) < 0;
    }

    public void exchange(@NotNull Comparable[] array, int index1, int index2) {
        Comparable temp = array[index1];
        array[index1] = array[index2];
        array[index2] = temp;
        count++;
    }

    public static void show(@NotNull Comparable[] a) {
//        Arrays.stream(a).forEach(System.out::println);
        System.out.println(Arrays.toString(a));
    }

    public static void compare(@NotNull Comparable[] a) {
//        Arrays.stream(a).forEach(System.out::println);
        System.out.println("原串" + Arrays.toString(al));
        System.out.println("现串" + Arrays.toString(a));
    }

    public static boolean isSorted(Comparable[] a) {
        for (int i = 1; i < a.length; i++) {
            if (less(a[i], a[i - 1])) {
                return false;
            }
        }
        return true;
    }

    public static Comparable[] getAl() {
        return al;
    }

    public int getCount() {
        return count;
    }

}

切面AOP

package uestc.zhangkx.utils;

import org.aspectj.lang.ProceedingJoinPoint;
import org.aspectj.lang.annotation.*;
import org.springframework.context.annotation.EnableAspectJAutoProxy;
import org.springframework.stereotype.Component;

@Component
@Aspect
@EnableAspectJAutoProxy
public class RunTime {

    @Pointcut(" execution(* uestc.zhangkx.sort.SortTemplate.sort(..))")
    private void pointcat() {
    }

    @Around("pointcat()")
    public Object aroundPrintLog(ProceedingJoinPoint pjp){
        Object rtValue = null;
        long startTime = 0;
        long endTime = 0;
        try{
            Object[] args = pjp.getArgs();
            startTime=System.currentTimeMillis();
            rtValue = pjp.proceed(args);
            endTime=System.currentTimeMillis();
            return rtValue;
        }catch (Throwable t){
            System.out.println("aop异常");
            throw new RuntimeException(t);
        }finally {
            System.out.print("RunTime->"+(endTime - startTime)+"ms");
        }
    }
}

 * @param a

@Component
public class SelectionSort extends SortTemplate {

    @Override
    public void sort(Comparable[] a) {
        int n = a.length;
        for (int i = 0; i < n; i++) {
            int min=i;
            for (int j = i; j < n; j++) {
                if (less(a[j],a[min])) {
                    min=j;
                }
            }
            exchange(a,min,i);
        }
    }

        show(al);
    }

}

希尔排序

package uestc.zhangkx.sort;

import org.junit.Test;
import org.springframework.stereotype.Component;

@Component
public class ShellSort extends SortTemplate {

    /**
     * 希尔排序
     * 希尔最佳序列[1,4,13.....]
     * @param a
     */
    @Override
    public void sort(Comparable[] a) {
        int n = a.length;
        int shell = 1;
        while(shell < n/3) {
            shell=3*shell+1;
        }
        while (shell>=1) {
            for (int i = shell; i < n; i++) {
                for (int j = i; j >= shell && less(a[j], a[j - shell]); j -= shell) {
                    exchange(a, j, j - shell);
                }
            }
            shell /= 3;
        }
    }

    @Test
    public void testSelSort() {
        show(al);
        sort(al);
        show(al);
    }
}

package uestc.zhangkx.sort;

import org.junit.Test;

/** 归并排序——自顶向下
 * @author zhangkx
 */
public class MergeDownSort extends SortTemplate {

    @Override
    public void sort(Comparable[] a) {
        aux=new Comparable[a.length];
        sort(a,0,a.length-1);
    }

    public void sort(Comparable[] a,int lo,int hi){
        if(hi<=lo){
            return ;
        }
        int mid = lo+(hi-lo)/2;
        sort(a,lo,mid);
        sort(a,mid+1,hi);
        merge(a,lo,mid,hi);
    }

    public static void merge(Comparable[] a,int lower,int middle,int higher){
        int i=lower,j=middle+1;
        for (int k = lower; k <= higher; k++) {
            aux[k]=a[k];
        }
        for (int k = lower; k <= higher; k++) {
            if (i>middle) {
                a[k]=aux[j++];
            } else if (j>higher) {
                a[k]=aux[i++];
            } else if (less(aux[j],aux[i])) {
                a[k]=aux[j++];
            } else {
                a[k]=aux[i++];
            }
        }
    }

    @Test
    public void testSelSort() {
        show(al);
        sort(al);
        show(al);
    }
}

自底向上

package uestc.zhangkx.sort;

import org.junit.Test;

/**
 * 归并排序——自底向上
 *
 * @author zhangkx
 */
public class MergeUpSort extends SortTemplate {

    @Override
    public void sort(Comparable[] a) {
        int n = a.length;
        aux = new Comparable[n];
        for (int size = 1; size < n; size<<=1) {
            for (int low = 0; low<n ; low+=size<<1) {
                merge(a,low,low+size-1,Math.min(low+size+size-1,n-1) );
            }
        }
    }

    public static void merge(Comparable[] a,int lower,int middle,int higher){
        int i=lower,j=middle+1;
        for (int k = lower; k <= higher; k++) {
            aux[k]=a[k];
        }
        for (int k = lower; k <= higher; k++) {
            if (i>middle) {
                a[k]=aux[j++];
            } else if (j>higher) {
                a[k]=aux[i++];
            } else if (less(aux[j],aux[i])) {
                a[k]=aux[j++];
            } else {
                a[k]=aux[i++];
            }
        }
    }

快速排序

package uestc.zhangkx.sort;

import org.junit.Test;

/**
 * 快速排序
 *
 * @author zhangkx
 */
public class QuickSort extends SortTemplate {
    @Override
    public void sort(Comparable[] a) {
        sort(a, 0, a.length - 1);
    }

    public void sort(Comparable[] a, int low, int high) {
        if (high <= low) {
            return;
        }
        int guard = partition(a, low, high);
        sort(a, low, guard - 1);
        sort(a, guard + 1, high);
    }

    private int partition(Comparable[] a, int low, int high) {
        int i = low, j = high + 1;
        Comparable v = a[low];
        while (true) {
            while (less(a[++i], v)) {
                if (i == high) {
                    break;
                }
            }
            while (less(v, a[--j])) {
                if (j == low) {
                    break;
                }
            }
            if (i >= j) {
                break;
            }
            exchange(a, i, j);
        }
        exchange(a, j, low);
        return j;
    }

    @Test
    public void test() {
        show(al);
        sort(al);
        show(al);
    }
}
